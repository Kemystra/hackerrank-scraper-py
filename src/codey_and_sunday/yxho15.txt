#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

vector<string> days = {"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"};

int main() {
    string a;
    int b;
    cin>>a;
    for(int i = 0; i< days.size(); i++){
        if(days[i] == a){
            b = 7 - (i+1);
            break;
        }
    }
    cout<<b;
    return 0;
}
