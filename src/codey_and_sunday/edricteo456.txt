def checkUntilSunday(day):
    days = {
        "monday" : 1,
        "tuesday" : 2,
        "wednesday" : 3,
        "thursday" : 4,
        "friday" : 5,
        "saturday" : 6,
        "sunday" : 7
    }
    
    dayValue = days[day.lower()]
    
    daysUntilSunday = 7 - dayValue
    
    return daysUntilSunday

s = input()
print(checkUntilSunday(s))
