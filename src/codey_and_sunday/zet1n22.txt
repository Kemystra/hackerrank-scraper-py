#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int daysss(const string& currentday){
    string days[] = {"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"};

int currentdayindex = 0;
for (int i=0;i<7;++i){
    if (days[i] == currentday){
        currentdayindex = i;
        break;
    }
}

int daysleft = (6-currentdayindex)%7;
return daysleft;
}

int main(){
    string currentday;
    cin>>currentday;
    
    cout<<daysss(currentday)<<endl;
    
    return 0;
}
