#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <unordered_map>
using namespace std;


int main() {
    string day;
    cin >> day;
    unordered_map<string, int> day_to_index = {
        {"Monday", 0},
        {"Tuesday", 1},
        {"Wednesday", 2},
        {"Thursday", 3},
        {"Friday", 4},
        {"Saturday", 5},
        {"Sunday", 6}
    };
    int current_day_index = day_to_index[day];
    int days_until_sunday = 6 - current_day_index;
    cout << days_until_sunday << endl;
    return 0;
}
