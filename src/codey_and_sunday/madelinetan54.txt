#include <iostream>
#include <string>
using namespace std;

int days_until_sunday(const string& day) {
    // Array of days from Monday to Saturday
    string days[] = {"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"};
    
    // Find the index of the current day
    int current_day_index = -1;
    for (int i = 0; i < 6; ++i) {
        if (days[i] == day) {
            current_day_index = i;
            break;
        }
    }
    
    // Calculate the days left to Sunday (index 6)
    int days_left = 6 - current_day_index;
    
    return days_left;
}

int main() {
    // Input the current day
    string current_day;
    cin >> current_day;
    
    // Output the number of days until the next Sunday
    cout << days_until_sunday(current_day) << endl;
    
    return 0;
}
