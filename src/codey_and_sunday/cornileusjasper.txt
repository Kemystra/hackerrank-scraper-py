def days_until_sunday(current_day):
    # Mapping of days to their respective positions in the week
    days = {
        "Monday": 1,
        "Tuesday": 2,
        "Wednesday": 3,
        "Thursday": 4,
        "Friday": 5,
        "Saturday": 6,
    }

    # Get the position of the current day
    current_position = days.get(current_day, 0)

    # Calculate days left until Sunday
    days_to_sunday = 7 - current_position

    return days_to_sunday

# Input
current_day = input().strip()

# Output
print(days_until_sunday(current_day))
